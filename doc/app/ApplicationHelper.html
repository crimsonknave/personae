def li_selected_if(condition, attributes = {}, &block)

  content_tag('li', attributes, &block)

selected_chronicle_id(user, session) click to toggle source

since users who aren’t logged in won’t have a current_user we need to stick a value in their session data to keep track of which chronicle they’re viewing.

# File app/helpers/application_helper.rb, line 7
def selected_chronicle_id(user, session)
  unless user.nil?
    user.selected_chronicle.id
  else
    session[:selected_chronicle_id] ||= Chronicle.first.id
    return session[:selected_chronicle_id]
  end
end
